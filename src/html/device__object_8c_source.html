349 static int resource_read(anjay_t *anjay,
350  const anjay_dm_object_def_t *const *obj_ptr,
351  anjay_iid_t iid,
352  anjay_rid_t rid,
353  anjay_riid_t riid,

358  assert(obj);
359  assert(iid == 0);
360

362

367  if(avst_get_free_memory(&memory_free)){
368  return -1;
369  }

371  device_object_add_error(&obj->def, OUT_OF_MEMORY);
372  }
373

376  }
377  case RID_ERROR_CODE:{

379  AVS_LIST_FOREACH(it, obj->error_codes){

390  avs_time_real_now())) {
391  return -1;
392  }
393  return anjay_ret_i64(ctx, timestamp);
394  }
395  case RID_SUPPORTED_BINDING_AND_MODES:{
396  assert(riid == ANJAY_ID_INVALID);
397  return anjay_ret_string(ctx, obj->supported_binding_and_modes);
398  }
399  case RID_MEMORY_TOTAL:{
400  assert(riid == ANJAY_ID_INVALID);
401  int32_t memory_total;
402  if(avst_get_total_memory(&memory_total)){
403  return -1;

406

411  }

577 int device_object_add_error(const anjay_dm_object_def_t **def, const error_code_t error_code){
578 
579  if (def) {
580  device_object_t *obj = get_obj(def);
581 
582  int32_t error_count = AVS_LIST_SIZE(obj->error_codes);
583 
584  error_code_instance_t new_error = {
585  .index = error_count,
586  .error_code = error_code
587  };
588 
589  error_code_instance_t* last_error = AVS_LIST_TAIL(obj->error_codes);
590 
591  //protection against adding NO_ERRO when other error items are in the list
592  if(error_count > 0 && last_error->error_code == NO_ERROR){
593  return -1;
594  }
595 
596  if(error_count == 1 && last_error->error_code == NO_ERROR){
597  new_error.index = 0;
598  *last_error = new_error;
599  return 0;
600  }
601  else{
602  if(find_error_instance(obj, error_code)){
603  return 0;
604  }
605  else{
606  return add_error_instance(obj, error_code) ? 0 : -1;
607  }
608  }
609  }
610  return -1;
611 }
